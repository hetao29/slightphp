  123.   /**
  124.    * Combination of public events - connect, login, disconnect
  125.    * @access public
  126.    * @param string $host 
  127.    * @param integer $port

  132.   public function Authorise ($host, $port = false, $tval = false, $username, $password, $debug_level = 0) {
  133.     $this->host = $host;
  134.  
  135.     //  If no port value is passed, retrieve it
  136.     if ($port == false) {
  137.       $this->port = $this->POP3_PORT;
  138.     } else {
  139.       $this->port = $port;
  140.     }
  141.  
  142.     //  If no port value is passed, retrieve it
  143.     if ($tval == false) {
  144.       $this->tval = $this->POP3_TIMEOUT;
  145.     } else {
  146.       $this->tval = $tval;
  147.     }
  148.  
  149.     $this->do_debug = $debug_level;
  150.     $this->username = $username;
  151.     $this->password = $password;
  152.  
  153.     //  Refresh the error log
  154.     $this->error = null;
  155.  
  156.     //  Connect
  157.     $result = $this->Connect($this->host, $this->port, $this->tval);
  158.  
  159.     if ($result) {
  160.       $login_result = $this->Login($this->username, $this->password);
  161.  
  162.       if ($login_result) {
  163.         $this->Disconnect();
  164.  
  165.         return true;
  166.       }
  167.  
  168.     }
  169.  
  170.     //  We need to disconnect regardless if the login succeeded
  171.     $this->Disconnect();
  172.  
  173.     return false;
  174.   }

  176.   /**
  177.    * Connect to the POP3 server
  178.    * @access public
  179.    * @param string $host 
  180.    * @param integer $port 
  181.    * @param integer $tval 
  182.    * @return boolean 
  183.    */
  184.   public function Connect ($host, $port = false, $tval = 30) {
  185.     //  Are we already connected?
  186.     if ($this->connected) {
  187.       return true;
  188.     }
  189.  
  190.     /*
  191.     On Windows this will raise a PHP Warning error if the hostname doesn't exist.
  192.     Rather than supress it with @fsockopen, let's capture it cleanly instead
  193.     */
  194.  
  195.     set_error_handler(array(&$this, 'catchWarning'));
  196.  
  197.     //  Connect to the POP3 server
  198.     $this->pop_conn = fsockopen($host,    //  POP3 Host
  199.                   $port,    //  Port #
  200.                   $errno,   //  Error Number
  201.                   $errstr,  //  Error Message
  202.                   $tval);   //  Timeout (seconds)
  203.  
  204.     //  Restore the error handler
  205.     restore_error_handler();
  206.  
  207.     //  Does the Error Log now contain anything?
  208.     if ($this->error && $this->do_debug >= 1) {
  209.       $this->displayErrors();
  210.     }
  211.  
  212.     //  Did we connect?
  213.     if ($this->pop_conn == false) {
  214.       //  It would appear not...
  215.       $this->error = array(
  216.         'error' => "Failed to connect to server $host on port $port",
  217.         'errno' => $errno,
  218.         'errstr' => $errstr
  219.       );
  220.  
  221.       if ($this->do_debug >= 1) {
  222.         $this->displayErrors();
  223.       }
  224.  
  225.       return false;
  226.     }
  227.  
  228.     //  Increase the stream time-out
  229.  
  230.     //  Check for PHP 4.3.0 or later
  231.     if (version_compare(phpversion(), '5.0.0', 'ge')) {
  232.       stream_set_timeout($this->pop_conn, $tval, 0);
  233.     } else {
  234.       //  Does not work on Windows
  235.       if (substr(PHP_OS, 0, 3) !== 'WIN') {
  236.         socket_set_timeout($this->pop_conn, $tval, 0);
  237.       }
  238.     }
  239.  
  240.     //  Get the POP3 server response
  241.     $pop3_response = $this->getResponse();
  242.  
  243.     //  Check for the +OK
  244.     if ($this->checkResponse($pop3_response)) {
  245.     //  The connection is established and the POP3 server is talking
  246.     $this->connected = true;
  247.       return true;
  248.     }
  249.  
  250.   }

  252.   /**
  253.    * Login to the POP3 server (does not support APOP yet)
  254.    * @access public
  255.    * @param string $username 
  256.    * @param string $password 
  257.    * @return boolean 
  258.    */
  259.   public function Login ($username = '', $password = '') {
  260.     if ($this->connected == false) {
  261.       $this->error = 'Not connected to POP3 server';
  262.  
  263.       if ($this->do_debug >= 1) {
  264.         $this->displayErrors();
  265.       }
  266.     }
  267.  
  268.     if (empty($username)) {
  269.       $username = $this->username;
  270.     }
  271.  
  272.     if (empty($password)) {
  273.       $password = $this->password;
  274.     }
  275.  
  276.     $pop_username = "USER $username" . $this->CRLF;
  277.     $pop_password = "PASS $password" . $this->CRLF;
  278.  
  279.     //  Send the Username
  280.     $this->sendString($pop_username);
  281.     $pop3_response = $this->getResponse();
  282.  
  283.     if ($this->checkResponse($pop3_response)) {
  284.       //  Send the Password
  285.       $this->sendString($pop_password);
  286.       $pop3_response = $this->getResponse();
  287.  
  288.       if ($this->checkResponse($pop3_response)) {
  289.         return true;
  290.       } else {
  291.         return false;
  292.       }
  293.     } else {
  294.       return false;
  295.     }
  296.   }

  299.    * Disconnect from the POP3 server

  302.   public function Disconnect () {
  303.     $this->sendString('QUIT');
  304.  
  305.     fclose($this->pop_conn);
  306.   }

  337.   /**

  343.   private function checkResponse ($string) {
  344.     if (substr($string, 0, 3) !== '+OK') {
  345.       $this->error = array(
  346.         'error' => "Server reported an error: $string",
  347.         'errno' => 0,
  348.         'errstr' => ''
  349.       );
  350.  
  351.       if ($this->do_debug >= 1) {
  352.         $this->displayErrors();
  353.       }
  354.  
  355.       return false;
  356.     } else {
  357.       return true;
  358.     }
  359.  
  360.   }

  367.     echo '<pre>';

  369.     foreach ($this->error as $single_error) {

  386.       'error' => "Connecting to the POP3 server raised a PHP warning: ",
